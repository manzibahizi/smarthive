# Smart Hive Solution - Docker Build & Push Commands
# Run these commands after Docker Desktop is installed

# 1. Build the Docker image
docker build -f Dockerfile.php -t smart-hive-solution:latest .

# 2. Tag for Docker Hub (replace 'yourusername' with your Docker Hub username)
docker tag smart-hive-solution:latest yourusername/smart-hive-solution:latest
docker tag smart-hive-solution:latest yourusername/smart-hive-solution:v1.0

# 3. Login to Docker Hub
docker login

# 4. Push to Docker Hub
docker push yourusername/smart-hive-solution:latest
docker push yourusername/smart-hive-solution:v1.0

# 5. Alternative: Build and run locally with docker-compose
docker-compose up --build -d

# 6. Check running containers
docker ps

# 7. View logs
docker-compose logs -f

# 8. Stop containers
docker-compose down
